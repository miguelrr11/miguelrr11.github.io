$validParents = @(
    "Algorithms",
    "Games",
    "IAs",
    "Math",
    "Physics_Simulations",
    "WebGL",
    "Automatas"
)

$results = @{}
$categoryTotals = @{}

git log --pretty=format:"%H" | ForEach-Object {
    $commit = $_

    git show --name-only --pretty=format: $commit |
    Where-Object { $_ -ne "" } |
    ForEach-Object {

        $parts = $_.Split("/")

        if ($parts.Length -eq 3 -and $validParents -contains $parts[0]) {

            $parent = $parts[0]
            $folder = $parts[1]

            # clave única por carpeta+commit
            $key = "$folder|$parent|$commit"
            $results[$key] = $true
        }
    }
}

# Agrupar por carpeta hija
$grouped = $results.Keys |
ForEach-Object {
    $parts = $_.Split("|")
    "$($parts[0])|$($parts[1])"
} |
Group-Object |
Sort-Object Count -Descending

Write-Host "==== COMMITS POR PROYECTO ===="

$totalGlobal = 0

foreach ($group in $grouped) {

    $split = $group.Name.Split("|")
    $folder = $split[0]
    $parent = $split[1]
    $count = $group.Count

    "{0,-6} {1,-20} {2,-20}" -f $count, $folder, $parent

    $totalGlobal += $count

    if ($categoryTotals.ContainsKey($parent)) {
        $categoryTotals[$parent] += $count
    } else {
        $categoryTotals[$parent] = $count
    }
}

Write-Host ""
Write-Host "==== TOTALES POR CATEGORIA ===="

foreach ($cat in $categoryTotals.GetEnumerator() | Sort-Object Value -Descending) {
    "{0,-6} {1}" -f $cat.Value, $cat.Key
}

Write-Host ""
Write-Host "==== TOTAL GLOBAL ===="
Write-Host $totalGlobal
